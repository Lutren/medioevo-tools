import { createContext, useContext, useEffect, useState, useRef, useCallback } from "react";
import { createRandom } from "../lib/prng";
import { GeodiaCell, PSIMetrics, computePSI, updateCell } from "../lib/geodia";
import { generateFingerprint } from "../lib/fingerprint";
import { scoreAllModelsFromPSI } from "../lib/physics";
import { useAppendWitnessEvent, useSaveSimulationSnapshot } from "@workspace/api-client-react";

type SimState = {
  isRunning: boolean;
  tick: number;
  seed: number;
  fingerprint: string;
  mode: string;
  truthModel: string;
  observerProfile: string;
  gridSize: number;
  cells: GeodiaCell[];
  psi: PSIMetrics;
};

type SimContextType = {
  state: SimState;
  start: () => void;
  pause: () => void;
  step: () => void;
  reset: (seed?: number) => void;
  setObserverProfile: (profile: string) => void;
  setTruthModel: (model: string) => void;
  setMode: (mode: string) => void;
};

const SimContext = createContext<SimContextType | null>(null);

export function SimulationProvider({ children }: { children: React.ReactNode }) {
  const appendWitness = useAppendWitnessEvent();
  const saveSnapshot  = useSaveSimulationSnapshot();

  const [state, setState] = useState<SimState>(() => {
    const initSeed = Math.floor(Math.random() * 10000);
    const initCells = initGrid(16, initSeed);
    return {
      isRunning: false,
      tick: 0,
      seed: initSeed,
      fingerprint: generateFingerprint(),
      mode: "SYNTHETIC_ONLY",
      truthModel: "null_flat",
      observerProfile: "instrument_balanced",
      gridSize: 16,
      cells: initCells,
      psi: computePSI(initCells, 0, 0),
    };
  });

  const prngRef = useRef(createRandom(state.seed));

  function initGrid(size: number, seed: number) {
    const prng = createRandom(seed);
    const cells: GeodiaCell[] = [];
    for (let y = 0; y < size; y++) {
      for (let x = 0; x < size; x++) {
        cells.push({
          x, y,
          resources:     prng.nextFloat(0, 1),
          knowledge:     prng.nextFloat(0, 0.5),
          memory:        0,
          conflict:      prng.nextFloat(0, 0.3),
          signal_noise:  prng.nextFloat(0, 1),
          stability:     0.5,
          agent_density: prng.nextFloat(0, 0.2),
        });
      }
    }
    return cells;
  }

  const doStep = useCallback(() => {
    setState(prev => {
      const nextCells = prev.cells.map(c => updateCell(c, prngRef.current.next));
      const nextTick  = prev.tick + 1;
      // Pass prevFatigue for exponential smoothing of epistemic fatigue
      const nextPsi   = computePSI(nextCells, nextTick, prev.psi.fatigue);

      if (nextTick % 10 === 0) {
        appendWitness.mutate({
          data: {
            event_type: "simulation_step",
            payload_summary: `tick=${nextTick} R=${nextPsi.R.toFixed(3)} Φeff=${nextPsi.phi_eff.toFixed(3)} regime=${nextPsi.regime} seed=${prev.seed} fp=${prev.fingerprint}`,
            mode: prev.mode as any,
            run_fingerprint: prev.fingerprint,
          }
        });
      }

      if (nextTick % 20 === 0) {
        const ranked = scoreAllModelsFromPSI(nextPsi, nextTick);
        const leader = ranked[0];
        saveSnapshot.mutate({
          data: {
            tick: nextTick,
            seed: prev.seed,
            run_fingerprint: prev.fingerprint,
            observer_profile: prev.observerProfile,
            truth_model: prev.truthModel,
            mode: prev.mode,
            psi_r:        nextPsi.R,
            psi_phi_eff:  nextPsi.phi_eff,
            psi_j_c:      nextPsi.j_c,
            psi_epsilon:  nextPsi.epsilon,
            psi_fatigue:  nextPsi.fatigue,
            psi_regime:   nextPsi.regime,
            eml_weight:   nextPsi.I_obs,   // I_obs used as EML proxy
            grid_summary: JSON.stringify({ cells: nextCells.length, R: nextPsi.R, sigma: nextPsi.sigma }),
            leading_model:       leader.name,
            leading_model_score: leader.score,
          }
        });
      }

      return { ...prev, tick: nextTick, cells: nextCells, psi: nextPsi };
    });
  }, [appendWitness, saveSnapshot]);

  useEffect(() => {
    if (!state.isRunning) return;
    const interval = setInterval(doStep, 500);
    return () => clearInterval(interval);
  }, [state.isRunning, doStep]);

  const start = () => setState(p => ({ ...p, isRunning: true }));
  const pause = () => setState(p => ({ ...p, isRunning: false }));
  const step  = () => { pause(); doStep(); };

  const reset = (newSeed?: number) => {
    const s = newSeed ?? Math.floor(Math.random() * 10000);
    prngRef.current = createRandom(s);
    const cells = initGrid(state.gridSize, s);
    setState(p => ({
      ...p,
      isRunning: false,
      tick: 0,
      seed: s,
      cells,
      psi: computePSI(cells, 0, 0),
    }));
  };

  const setObserverProfile = (p: string) => setState(s => ({ ...s, observerProfile: p }));
  const setTruthModel      = (m: string) => setState(s => ({ ...s, truthModel: m }));
  const setMode            = (m: string) => setState(s => ({ ...s, mode: m }));

  return (
    <SimContext.Provider value={{ state, start, pause, step, reset, setObserverProfile, setTruthModel, setMode }}>
      {children}
    </SimContext.Provider>
  );
}

export function useGeodiaSimulation() {
  const ctx = useContext(SimContext);
  if (!ctx) throw new Error("Missing SimulationProvider");
  return ctx;
}
