import { Router, type IRouter } from "express";
import { db, simulationSnapshotsTable } from "@workspace/db";
import { desc } from "drizzle-orm";
import { SaveSimulationSnapshotBody, ListSimulationSnapshotsQueryParams } from "@workspace/api-zod";

const router: IRouter = Router();

router.get("/simulation/snapshots", async (req, res): Promise<void> => {
  const query = ListSimulationSnapshotsQueryParams.safeParse(req.query);
  if (!query.success) {
    res.status(400).json({ error: query.error.message });
    return;
  }
  const limit = query.data.limit ?? 20;
  const rows = await db
    .select()
    .from(simulationSnapshotsTable)
    .orderBy(desc(simulationSnapshotsTable.created_at))
    .limit(limit);
  res.json(rows);
});

router.post("/simulation/snapshots", async (req, res): Promise<void> => {
  const parsed = SaveSimulationSnapshotBody.safeParse(req.body);
  if (!parsed.success) {
    res.status(400).json({ error: parsed.error.message });
    return;
  }
  const [row] = await db.insert(simulationSnapshotsTable).values(parsed.data).returning();
  res.status(201).json(row);
});

router.get("/simulation/snapshots/latest", async (_req, res): Promise<void> => {
  const [row] = await db
    .select()
    .from(simulationSnapshotsTable)
    .orderBy(desc(simulationSnapshotsTable.created_at))
    .limit(1);
  if (!row) {
    res.status(404).json({ error: "No snapshots found" });
    return;
  }
  res.json(row);
});

export default router;
