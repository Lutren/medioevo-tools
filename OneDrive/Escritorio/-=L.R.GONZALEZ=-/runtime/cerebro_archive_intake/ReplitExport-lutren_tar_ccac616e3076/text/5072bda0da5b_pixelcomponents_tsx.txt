import { cn } from "@/lib/utils";

export function PixelBorder({ children, className, interactive = false, ...props }: React.HTMLAttributes<HTMLDivElement> & { interactive?: boolean }) {
  return (
    <div 
      className={cn("pixel-border p-3", interactive && "pixel-border-interactive cursor-pointer", className)} 
      {...props}
    >
      {children}
    </div>
  );
}

export function PixelButton({ children, className, ...props }: React.ButtonHTMLAttributes<HTMLButtonElement>) {
  return (
    <button 
      className={cn("pixel-border pixel-border-interactive px-4 py-2 font-display font-bold uppercase tracking-wider text-primary active:scale-95 transition-transform", className)}
      {...props}
    >
      {children}
    </button>
  );
}

export function ModeBadge({ mode, className }: { mode: string, className?: string }) {
  const isRed = mode === "BLOCK" || mode === "REVIEW_REQUIRED";
  return (
    <span className={cn(
      "inline-block px-2 py-0.5 text-xs font-mono font-bold border border-current",
      isRed ? "text-destructive border-destructive" : "text-primary border-primary",
      className
    )}>
      {mode}
    </span>
  );
}

export function EpistLabel({ label, className }: { label: string, className?: string }) {
  return (
    <span className={cn(
      "inline-block px-1.5 py-0.5 text-[10px] font-mono border border-current",
      label === "BASELINE NULO" || label.includes("WEAK") ? "text-destructive" :
      label === "REVIEW_REQUIRED" ? "text-primary" : "text-success",
      className
    )}>
      {label}
    </span>
  );
}
