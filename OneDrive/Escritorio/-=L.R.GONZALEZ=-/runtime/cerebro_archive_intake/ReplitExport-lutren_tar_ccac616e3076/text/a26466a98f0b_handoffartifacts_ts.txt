import { pgTable, serial, text, timestamp, integer } from "drizzle-orm/pg-core";
import { createInsertSchema } from "drizzle-zod";
import { z } from "zod/v4";

export const handoffArtifactsTable = pgTable("handoff_artifacts", {
  id: serial("id").primaryKey(),
  created_at: timestamp("created_at", { withTimezone: true }).notNull().defaultNow(),
  estado: text("estado").notNull(),
  certeza: text("certeza").notNull(),
  inferencia: text("inferencia").notNull(),
  incognita: text("incognita").notNull(),
  accion_siguiente: text("accion_siguiente").notNull(),
  blockers: text("blockers").notNull(),
  fingerprint: text("fingerprint").notNull(),
  tick: integer("tick").notNull(),
  seed: integer("seed").notNull(),
  observer_profile: text("observer_profile").notNull(),
  mode: text("mode").notNull(),
});

export const insertHandoffArtifactSchema = createInsertSchema(handoffArtifactsTable).omit({ id: true, created_at: true });
export type InsertHandoffArtifact = z.infer<typeof insertHandoffArtifactSchema>;
export type HandoffArtifactRow = typeof handoffArtifactsTable.$inferSelect;
