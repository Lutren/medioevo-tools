import { pgTable, serial, text, timestamp } from "drizzle-orm/pg-core";
import { createInsertSchema } from "drizzle-zod";
import { z } from "zod/v4";

export const witnessEventsTable = pgTable("witness_events", {
  id: serial("id").primaryKey(),
  timestamp: timestamp("timestamp", { withTimezone: true }).notNull().defaultNow(),
  event_type: text("event_type").notNull(),
  payload_summary: text("payload_summary").notNull(),
  hash: text("hash").notNull(),
  previous_hash: text("previous_hash").notNull(),
  mode: text("mode").notNull(),
  run_fingerprint: text("run_fingerprint").notNull(),
});

export const insertWitnessEventSchema = createInsertSchema(witnessEventsTable).omit({ id: true, timestamp: true });
export type InsertWitnessEvent = z.infer<typeof insertWitnessEventSchema>;
export type WitnessEventRow = typeof witnessEventsTable.$inferSelect;
