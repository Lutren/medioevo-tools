import { Router, type IRouter } from "express";
import { db, witnessEventsTable } from "@workspace/db";
import { desc, eq } from "drizzle-orm";
import crypto from "crypto";
import {
  AppendWitnessEventBody,
  ListWitnessEventsQueryParams,
  GetRecentWitnessEventsQueryParams,
} from "@workspace/api-zod";

const router: IRouter = Router();

function sha256(data: string): string {
  return crypto.createHash("sha256").update(data).digest("hex");
}

router.get("/witness", async (req, res): Promise<void> => {
  const query = ListWitnessEventsQueryParams.safeParse(req.query);
  if (!query.success) {
    res.status(400).json({ error: query.error.message });
    return;
  }
  const { limit, event_type } = query.data;
  let rows = await db
    .select()
    .from(witnessEventsTable)
    .orderBy(desc(witnessEventsTable.timestamp))
    .limit(limit ?? 50);

  if (event_type) {
    rows = rows.filter((r) => r.event_type === event_type);
  }

  res.json(rows);
});

router.get("/witness/recent", async (req, res): Promise<void> => {
  const query = GetRecentWitnessEventsQueryParams.safeParse(req.query);
  if (!query.success) {
    res.status(400).json({ error: query.error.message });
    return;
  }
  const n = query.data.n ?? 10;
  const rows = await db
    .select()
    .from(witnessEventsTable)
    .orderBy(desc(witnessEventsTable.timestamp))
    .limit(n);
  res.json(rows);
});

router.post("/witness", async (req, res): Promise<void> => {
  const parsed = AppendWitnessEventBody.safeParse(req.body);
  if (!parsed.success) {
    res.status(400).json({ error: parsed.error.message });
    return;
  }

  const lastRows = await db
    .select()
    .from(witnessEventsTable)
    .orderBy(desc(witnessEventsTable.id))
    .limit(1);

  const previousHash = lastRows.length > 0 ? lastRows[0].hash : "GENESIS_BLOCK_000000000000000000000000000000000000000";

  const payload = {
    ...parsed.data,
    timestamp: new Date().toISOString(),
    previous_hash: previousHash,
  };
  const hash = sha256(JSON.stringify(payload));

  const [row] = await db
    .insert(witnessEventsTable)
    .values({ ...parsed.data, hash, previous_hash: previousHash })
    .returning();

  res.status(201).json(row);
});

export default router;
