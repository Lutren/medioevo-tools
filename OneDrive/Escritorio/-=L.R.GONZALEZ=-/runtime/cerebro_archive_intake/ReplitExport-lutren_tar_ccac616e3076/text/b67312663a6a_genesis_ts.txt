import { Router, type IRouter } from "express";
import { db, genesisConfigsTable } from "@workspace/db";
import { desc } from "drizzle-orm";
import { SaveGenesisConfigBody } from "@workspace/api-zod";

const router: IRouter = Router();

router.get("/genesis", async (_req, res): Promise<void> => {
  const [row] = await db
    .select()
    .from(genesisConfigsTable)
    .orderBy(desc(genesisConfigsTable.created_at))
    .limit(1);

  if (!row) {
    res.status(404).json({ error: "No genesis config found" });
    return;
  }
  res.json(row);
});

router.post("/genesis", async (req, res): Promise<void> => {
  const parsed = SaveGenesisConfigBody.safeParse(req.body);
  if (!parsed.success) {
    res.status(400).json({ error: parsed.error.message });
    return;
  }
  const [row] = await db.insert(genesisConfigsTable).values(parsed.data).returning();
  res.status(201).json(row);
});

export default router;
