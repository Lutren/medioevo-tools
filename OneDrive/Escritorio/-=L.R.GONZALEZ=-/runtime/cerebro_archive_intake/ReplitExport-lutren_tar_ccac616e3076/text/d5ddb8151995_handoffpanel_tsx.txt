import { useState, useEffect } from "react";
import { useGeodiaSimulation } from "@/contexts/SimulationContext";
import { useSaveHandoff, useListHandoffs } from "@workspace/api-client-react";
import { PixelBorder, PixelButton } from "./custom/PixelComponents";
import { useToast } from "@/hooks/use-toast";

export default function HandoffPanel() {
  const { state } = useGeodiaSimulation();
  const saveHandoff = useSaveHandoff();
  const { data: list = [], refetch } = useListHandoffs();
  const { toast } = useToast();

  const [form, setForm] = useState({
    estado: "",
    certeza: "MEDIA",
    inferencia: "",
    incognita: "",
    accion_siguiente: "",
    blockers: ""
  });

  const handleSave = () => {
    saveHandoff.mutate({
      data: {
        ...form,
        fingerprint: state.fingerprint,
        tick: state.tick,
        seed: state.seed,
        observer_profile: state.observerProfile,
        mode: state.mode
      }
    }, {
      onSuccess: () => {
        toast({ title: "[HANDOFF GUARDADO]", description: "Sincronizado con BD local." });
        refetch();
        setForm(f => ({ ...f, estado: "", inferencia: "", incognita: "", accion_siguiente: "", blockers: "" }));
      }
    });
  };

  const handleExport = () => {
    const text = `
=== DUAT GEODIA HANDOFF ===
FINGERPRINT: ${state.fingerprint}
TICK: ${state.tick} | SEED: ${state.seed}
MODE: ${state.mode}
===========================
ESTADO:
${form.estado}

CERTEZA: ${form.certeza}

INFERENCIA:
${form.inferencia}

INCÓGNITA:
${form.incognita}

ACCIÓN SIGUIENTE:
${form.accion_siguiente}

BLOCKERS:
${form.blockers}
    `.trim();
    navigator.clipboard.writeText(text);
    toast({ title: "[COPIADO AL PORTAPAPELES]", description: "Listo para transmisión externa." });
  };

  return (
    <div className="flex flex-col h-full overflow-y-auto pb-4 pr-2 space-y-4">
      <div className="font-display text-xs border-b border-border pb-1 font-bold">
        HANDOFF // EXPORTAR BRIEF
      </div>

      <div className="space-y-2 font-mono text-[10px]">
        <div className="grid grid-cols-2 gap-2 text-primary/70 bg-black/50 p-2 border border-border border-dashed">
          <div>FP: {state.fingerprint}</div>
          <div>TICK: {state.tick}</div>
          <div>SEED: {state.seed}</div>
          <div>MODE: {state.mode}</div>
        </div>

        <div className="space-y-1 mt-2">
          <label>ESTADO:</label>
          <textarea value={form.estado} onChange={e => setForm(f => ({...f, estado: e.target.value}))} className="w-full h-12 bg-input border border-border text-primary p-1 focus:outline-none resize-none" />
        </div>

        <div className="space-y-1">
          <label>CERTEZA:</label>
          <input value={form.certeza} onChange={e => setForm(f => ({...f, certeza: e.target.value}))} className="w-full bg-input border border-border text-primary p-1 focus:outline-none" />
        </div>

        <div className="space-y-1">
          <label>INFERENCIA:</label>
          <textarea value={form.inferencia} onChange={e => setForm(f => ({...f, inferencia: e.target.value}))} className="w-full h-12 bg-input border border-border text-primary p-1 focus:outline-none resize-none" />
        </div>

        <div className="space-y-1">
          <label>INCÓGNITA:</label>
          <textarea value={form.incognita} onChange={e => setForm(f => ({...f, incognita: e.target.value}))} className="w-full h-12 bg-input border border-border text-primary p-1 focus:outline-none resize-none" />
        </div>

        <div className="space-y-1">
          <label>ACCIÓN SIGUIENTE:</label>
          <textarea value={form.accion_siguiente} onChange={e => setForm(f => ({...f, accion_siguiente: e.target.value}))} className="w-full h-12 bg-input border border-border text-primary p-1 focus:outline-none resize-none" />
        </div>

        <div className="space-y-1">
          <label>BLOCKERS:</label>
          <textarea value={form.blockers} onChange={e => setForm(f => ({...f, blockers: e.target.value}))} className="w-full h-12 bg-input border border-border text-primary p-1 focus:outline-none resize-none" />
        </div>

        <div className="flex gap-2 pt-2">
          <PixelButton className="flex-1 text-[10px] py-1" onClick={handleSave} disabled={saveHandoff.isPending}>[GUARDAR]</PixelButton>
          <PixelButton className="flex-1 text-[10px] py-1 border-accent text-accent" onClick={handleExport}>[EXPORTAR]</PixelButton>
        </div>
      </div>

      {list.length > 0 && (
        <div className="mt-4 border-t border-border pt-4 space-y-2">
          <div className="text-[10px] font-display font-bold">HISTORIAL DE HANDOFFS</div>
          {list.map(h => (
            <PixelBorder key={h.id} className="p-2 cursor-pointer hover:bg-input">
              <div className="flex justify-between text-[9px] font-mono text-primary/70">
                <span>{new Date(h.created_at).toLocaleDateString()}</span>
                <span>TICK {h.tick}</span>
              </div>
              <div className="font-mono text-[10px] truncate mt-1">{h.estado}</div>
            </PixelBorder>
          ))}
        </div>
      )}
    </div>
  );
}
