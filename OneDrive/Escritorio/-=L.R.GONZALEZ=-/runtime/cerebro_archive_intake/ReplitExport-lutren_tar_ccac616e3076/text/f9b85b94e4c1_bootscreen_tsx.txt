// ─── OSIT Digital Rain Characters ─────────────────────────────────────────────
const RAIN_CHARS = "ΦεΣΠμν∂∇⬡◈Rε01Jc∑OSIT⌬◆▓░∞≡⟨⟩01ΦΦεε10".split("");

function DigitalRain() {
  const canvasRef = useRef<HTMLCanvasElement>(null);

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext("2d");
    if (!ctx) return;

    const resize = () => {
      const rect = canvas.parentElement?.getBoundingClientRect();
      canvas.width  = rect?.width  ?? window.innerWidth;
      canvas.height = rect?.height ?? window.innerHeight;
    };
    resize();

    const CHAR_W   = 16;
    const CHAR_H   = 18;
    const cols     = Math.ceil(canvas.width / CHAR_W);
    const drops    = Array.from({ length: cols }, () => -Math.random() * 30);
    const speeds   = Array.from({ length: cols }, () => 0.25 + Math.random() * 0.55);
    // Assign each column a hue: 80% cyan, 10% magenta, 10% green
    const hues     = Array.from({ length: cols }, () => {
      const r = Math.random();
      if (r < 0.80) return "cyan";
      if (r < 0.90) return "magenta";
      return "green";
    });

    let raf: number;
    let last = 0;

    const COLORS: Record<string, [number, number, number]> = {
      cyan:    [0,   240, 255],
      magenta: [255, 0,   184],
      green:   [0,   255, 136],
    };

    const draw = (t: number) => {
      raf = requestAnimationFrame(draw);
      if (t - last < 55) return;           // ~18 fps
      last = t;

      ctx.fillStyle = "rgba(2, 6, 9, 0.065)";
      ctx.fillRect(0, 0, canvas.width, canvas.height);

      ctx.font = `${CHAR_H - 4}px 'Share Tech Mono', monospace`;

      for (let i = 0; i < cols; i++) {
        const y = drops[i] * CHAR_H;
        if (y < -CHAR_H) { drops[i] += speeds[i]; continue; }

        const char = RAIN_CHARS[Math.floor(Math.random() * RAIN_CHARS.length)];
        const [r, g, b] = COLORS[hues[i]];
        const frac = drops[i] - Math.floor(drops[i]);

        // Lead char: bright white
        if (frac < 0.15) {
          ctx.fillStyle = "#e8faff";
        } else {
          const alpha = 0.15 + Math.random() * 0.70;
          ctx.fillStyle = `rgba(${r},${g},${b},${alpha})`;
        }

        ctx.fillText(char, i * CHAR_W, y);

        drops[i] += speeds[i];
        if (y > canvas.height && Math.random() > 0.975) {
          drops[i] = -8 - Math.random() * 25;
        }
      }
    };

    raf = requestAnimationFrame(draw);
    return () => cancelAnimationFrame(raf);
  }, []);

  return (
    <canvas
      ref={canvasRef}
      className="absolute inset-0 w-full h-full pointer-events-none"
      style={{ opacity: 0.38 }}
    />
  );
}
